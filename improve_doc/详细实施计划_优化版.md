```python
# backend/app/services/paper_service.py
class PaperService:
    async def upload_paper(
        self, 
        file: UploadFile, 
        user_id: int
    ) -> Paper:
        """上传并处理PDF"""
        # 1. 保存文件到MinIO
        pdf_url = await self.storage.upload(file)
        
        # 2. 创建数据库记录
        paper = await self.repo.create(
            user_id=user_id,
            pdf_url=pdf_url,
            status="processing"
        )
        
        # 3. 启动异步解析任务
        parse_pdf_task.delay(paper.id)
        
        return paper
```

#### PDF解析任务

```python
# backend/app/tasks/pdf_tasks.py
@celery_app.task(bind=True, max_retries=3)
def parse_pdf_task(self, paper_id: int):
    """异步解析PDF"""
    try:
        # 1. 下载PDF
        pdf_path = download_pdf(paper_id)
        
        # 2. 解析文本
        parser = PDFParser()
        result = parser.parse(pdf_path)
        
        # 3. 提取元数据(使用LLM)
        metadata = extract_metadata_with_llm(result['text'])
        
        # 4. 向量化并存储
        embeddings = bge_model.encode(result['chunks'])
        milvus.insert(paper_id, embeddings)
        
        # 5. 更新状态
        update_paper_status(paper_id, "completed", metadata)
        
    except Exception as e:
        self.retry(exc=e, countdown=60)
```

### Week 13-14: NLP分析功能

#### 关键词提取服务

```python
# backend/app/services/nlp_service.py
from FlagEmbedding import BGEM3FlagModel

class NLPService:
    def __init__(self):
        self.bge_model = BGEM3FlagModel('BAAI/bge-m3')
    
    def extract_keywords(self, text: str, top_k: int = 20):
        """混合关键词提取"""
        # 1. TF-IDF
        tfidf_kws = self._tfidf_extract(text, top_k)
        
        # 2. TextRank
        textrank_kws = self._textrank_extract(text, top_k)
        
        # 3. 语义关键词 (使用BGE)
        semantic_kws = self._semantic_extract(text, top_k)
        
        # 4. 融合排序
        return self._merge_keywords(tfidf_kws, textrank_kws, semantic_kws)
    
    def encode_dense(self, text: str):
        """生成稠密向量"""
        return self.bge_model.encode(
            text, 
            return_dense=True,
            return_sparse=False
        )['dense_vecs']
    
    def encode_sparse(self, text: str):
        """生成稀疏向量"""
        return self.bge_model.encode(
            text,
            return_dense=False,
            return_sparse=True
        )['lexical_weights']
```

```python
# backend/app/rag/retriever.py
from typing import List
from langchain.schema import Document

class HybridRetriever:
    """三路召回混合检索器"""
    
    def __init__(self):
        self.es_client = Elasticsearch()
        self.milvus_client = MilvusClient()
        self.bge_model = BGEM3FlagModel('BAAI/bge-m3')
    
    async def retrieve(
        self, 
        query: str, 
        top_k: int = 10,
        project_id: int = None
    ) -> List[Document]:
        """混合检索"""
        
        # 1. BM25全文检索
        bm25_results = await self._bm25_search(query, top_k * 3, project_id)
        
        # 2. 稠密向量检索
        query_dense = self.bge_model.encode(query, return_dense=True)
        dense_results = await self._dense_search(query_dense, top_k * 3, project_id)
        
        # 3. 稀疏向量检索
        query_sparse = self.bge_model.encode(query, return_sparse=True)
        sparse_results = await self._sparse_search(query_sparse, top_k * 3, project_id)
        
        # 4. RRF融合排序
        fused = self._rrf_fusion([
            (bm25_results, 0.3),
            (dense_results, 0.5),
            (sparse_results, 0.2)
        ])
        
        # 5. 重排序 (可选)
        reranked = await self._rerank(query, fused[:top_k * 2])
        
        return reranked[:top_k]
    
    def _rrf_fusion(self, results_with_weights):
        """Reciprocal Rank Fusion"""
        doc_scores = {}
        k = 60  # RRF参数
        
        for results, weight in results_with_weights:
            for rank, doc in enumerate(results):
                doc_id = doc.metadata['id']
                if doc_id not in doc_scores:
                    doc_scores[doc_id] = {'doc': doc, 'score': 0}
                doc_scores[doc_id]['score'] += weight * (1 / (k + rank))
        
        sorted_docs = sorted(
            doc_scores.values(), 
            key=lambda x: x['score'], 
            reverse=True
        )
        return [item['doc'] for item in sorted_docs]
```

```python
import numpy as np
from scipy import stats

class TrendAnalyzer:
    def detect_burst_keywords(
        self, 
        keyword_time_series: dict,
        window_size: int = 3
    ) -> List[dict]:
        """检测突现关键词"""
        bursting_keywords = []
        
        for keyword, counts in keyword_time_series.items():
            if len(counts) < window_size + 1:
                continue
            
            # 计算增长率
            recent = np.mean(counts[-window_size:])
            historical = np.mean(counts[:-window_size])
            
            if historical > 0:
                growth_rate = (recent - historical) / historical
                
                # 使用Kleinberg方法检测突现
                is_bursting = self._kleinberg_burst_detection(counts)
                
                if is_bursting or growth_rate > 1.0:  # 增长超过100%
                    bursting_keywords.append({
                        "keyword": keyword,
                        "growth_rate": growth_rate,
                        "is_bursting": is_bursting,
                        "recent_count": recent,
                        "historical_count": historical
                    })
        
        return sorted(bursting_keywords, key=lambda x: x['growth_rate'], reverse=True)
```

```python
    async def detect_gaps(
        self, 
        topic: str,
        papers: List[dict]
    ) -> List[dict]:
        """检测研究空白"""
        
        # 1. 提取所有关键概念
        concepts = await self._extract_concepts(papers)
        
        # 2. 构建概念共现矩阵
        cooccurrence = self._build_cooccurrence_matrix(concepts, papers)

        clustering = DBSCAN(eps=0.3, min_samples=2).fit(embeddings)
        
        # 4. 识别稀疏区域 (研究空白)
        gaps = []
        for i, concept in enumerate(concepts):
            if clustering.labels_[i] == -1:  # 噪声点 = 孤立概念
                gaps.append({
                    "concept": concept["name"],
                    "type": "under_researched",

                })

        cross_domain_gaps = self._find_cross_domain_opportunities(
            clusters=clustering,
            concepts=concepts,
            cooccurrence=cooccurrence
        )
        gaps.extend(cross_domain_gaps)
        
        # 6. 使用LLM生成研究建议
        suggestions = await self._generate_suggestions(gaps, topic)
```

```python
                            "image_url": {
                                "url": f"data:image/png;base64,{image_data}"
                            }
                        }
                    ]
                }
            ]
        )
        
        return {
            "analysis": response.choices[0].message.content,
            "image_path": image_path
        }
    
    async def extract_formulas(self, pdf_path: str) -> List[dict]:
        """提取和解释公式"""
        # 使用MathPix或自定义OCR识别公式
        formulas = await self._ocr_formulas(pdf_path)
        
        explanations = []
        for formula in formulas:
            explanation = await self.llm.generate(
                f"请解释这个学术公式的含义：{formula['latex']}"
            )
            explanations.append({
                "latex": formula["latex"],
                "location": formula["page"],
                "explanation": explanation
            })
        
        return explanations
```

```typescript
// browser-extension/src/content.ts
// Chrome Extension - Content Script

class PaperAnalyzerExtension {
  private apiUrl = 'https://api.paperbigdata.com';
  
  async detectPaper(): Promise<PaperInfo | null> {
    const url = window.location.href;
    
    // 检测常见学术网站
    if (url.includes('arxiv.org')) {
      return this.parseArxiv();
    } else if (url.includes('doi.org')) {
      return this.parseDOI();
    } else if (url.includes('scholar.google')) {
      return this.parseGoogleScholar();
    }
    
    return null;
  }
  
  private parseArxiv(): PaperInfo {
    const title = document.querySelector('.title')?.textContent;
    const abstract = document.querySelector('.abstract')?.textContent;
    const arxivId = window.location.pathname.split('/').pop();
    
    return {
      title: title?.replace('Title:', '').trim(),
      abstract: abstract?.replace('Abstract:', '').trim(),
      source: 'arxiv',
      sourceId: arxivId,
      pdfUrl: `https://arxiv.org/pdf/${arxivId}.pdf`
    };
  }
  
  async savePaper(paperInfo: PaperInfo): Promise<void> {
    const token = await this.getAuthToken();
    
    await fetch(`${this.apiUrl}/v1/papers/save-from-extension`, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${token}`,
        'Content-Type': 'application/json'
      },
      body: JSON.stringify(paperInfo)
    });
    
    this.showNotification('论文已保存到您的文献库');
  }
  
  async quickAnalyze(paperInfo: PaperInfo): Promise<void> {
    // 打开侧边栏显示快速分析结果
    const analysis = await this.fetchAnalysis(paperInfo);
    this.renderSidebar(analysis);
  }
}
```
