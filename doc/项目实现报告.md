### 1.2 技术选型理由

#### 前端: React + Ant Design + ECharts
- **React**: 组件化开发,生态丰富,性能优秀
- **Ant Design**: 企业级UI组件,开箱即用
- **ECharts**: 强大的可视化能力,支持多种图表

#### 后端: FastAPI + Python
- **FastAPI**: 高性能,自动生成API文档,类型提示
- **Python**: NLP生态成熟,丰富的科学计算库

#### 数据库组合
- **PostgreSQL**: 可靠的ACID事务,强大的查询能力
- **MongoDB**: 灵活的文档存储,适合非结构化数据
- **Redis**: 高性能缓存,消息队列
- **Elasticsearch**: 强大的全文检索和聚合分析
- **Milvus**: 专业的向量数据库,支持相似度检索

### 1.3 系统特性

#### 高可用性
- 服务无状态设计,支持水平扩展
- 数据库主从复制,读写分离
- Redis哨兵模式,自动故障转移

#### 高性能

- 多级缓存策略(浏览器缓存、CDN、Redis)

#### 安全性

### 3.1 PDF解析模块

#### 实现架构
```python
# backend/services/pdf_parser.py

import pdfplumber
import pytesseract
from pdf2image import convert_from_path
from typing import Dict, List
import re

class PDFParser:
    """PDF解析器"""
    
    def __init__(self):
        self.supported_languages = ['chi_sim', 'eng']
    
    def parse_pdf(self, pdf_path: str) -> Dict:
        """
        解析PDF文件
        
        Args:
            pdf_path: PDF文件路径
            
        Returns:
            解析结果字典
        """
        try:
            # 尝试文本型PDF解析
            result = self._parse_text_pdf(pdf_path)
            
            # 如果文本提取失败,使用OCR
            if not result['full_text'] or len(result['full_text']) < 100:
                result = self._parse_image_pdf(pdf_path)
            
            # 提取元数据
            result['metadata'] = self._extract_metadata(result['full_text'])
            
            return result
            
        except Exception as e:
            raise Exception(f"PDF解析失败: {str(e)}")
    
    def _parse_text_pdf(self, pdf_path: str) -> Dict:
        """解析文本型PDF"""
        result = {
            'full_text': '',
            'sections': [],
            'page_count': 0
        }
        
        with pdfplumber.open(pdf_path) as pdf:
            result['page_count'] = len(pdf.pages)
            
            for page_num, page in enumerate(pdf.pages, 1):
                text = page.extract_text()
                if text:
                    result['full_text'] += text + '\n'
                    result['sections'].append({
                        'page': page_num,
                        'content': text
                    })
        
        return result
    
    def _parse_image_pdf(self, pdf_path: str) -> Dict:
        """使用OCR解析扫描型PDF"""
        result = {
            'full_text': '',
            'sections': [],
            'page_count': 0
        }
        
        # 转换PDF为图像
        images = convert_from_path(pdf_path)
        result['page_count'] = len(images)
        
        for page_num, image in enumerate(images, 1):
            # OCR识别
            text = pytesseract.image_to_string(
                image,
                lang='+'.join(self.supported_languages)
            )
            result['full_text'] += text + '\n'
            result['sections'].append({
                'page': page_num,
                'content': text
            })
        
        return result
    
    def _extract_metadata(self, text: str) -> Dict:
        """提取元数据"""
        metadata = {
            'title': self._extract_title(text),
            'authors': self._extract_authors(text),
            'abstract': self._extract_abstract(text),
            'keywords': self._extract_keywords(text),
            'references': self._extract_references(text)
        }
        return metadata
    
    def _extract_title(self, text: str) -> str:
        """提取标题(取前100个字符中的第一行)"""
        lines = text.split('\n')
        for line in lines[:5]:
            line = line.strip()
            if len(line) > 10:
                return line
        return "Untitled"
    
    def _extract_authors(self, text: str) -> List[str]:
        """提取作者"""
        # 简单实现:查找常见作者模式
        author_pattern = r'([A-Z][a-z]+\s+[A-Z][a-z]+)(?:,|\s+and\s+)'
        authors = re.findall(author_pattern, text[:1000])
        return list(set(authors))[:10]  # 最多10个作者
    
    def _extract_abstract(self, text: str) -> str:
        """提取摘要"""
        # 查找Abstract关键词
        abstract_pattern = r'(?:Abstract|ABSTRACT|摘要)[:\s]+(.*?)(?:\n\n|Keywords|关键词|Introduction|1\.)'
        match = re.search(abstract_pattern, text, re.DOTALL | re.IGNORECASE)
        if match:
            return match.group(1).strip()[:1000]  # 最多1000字符
        return ""
    
    def _extract_keywords(self, text: str) -> List[str]:
        """提取关键词"""
        # 查找Keywords部分
        keywords_pattern = r'(?:Keywords|关键词)[:\s]+(.*?)(?:\n\n|Introduction|1\.)'
        match = re.search(keywords_pattern, text, re.IGNORECASE)
        if match:
            keywords_text = match.group(1)
            # 分割关键词
            keywords = re.split(r'[;,，；]', keywords_text)
            return [k.strip() for k in keywords if k.strip()][:10]
        return []
    
    def _extract_references(self, text: str) -> List[str]:
        """提取参考文献"""
        # 查找References部分
        ref_pattern = r'(?:References|参考文献|REFERENCES)(.*?)$'
        match = re.search(ref_pattern, text, re.DOTALL | re.IGNORECASE)
        if match:
            ref_text = match.group(1)
            # 分割单个参考文献(通常以[1]、[2]或换行开始)
            refs = re.split(r'\[\d+\]|\n(?=\[)', ref_text)
            return [r.strip() for r in refs if r.strip()][:50]  # 最多50条
        return []
```

### 3.2 NLP分析模块

#### 关键词提取实现
```python
# backend/services/nlp_analyzer.py

import jieba
import jieba.analyse
from sklearn.feature_extraction.text import TfidfVectorizer
from transformers import AutoTokenizer, AutoModel
import torch
import numpy as np
from typing import List, Dict

class NLPAnalyzer:
    """NLP分析器"""
    
    def __init__(self):
        # 加载停用词
        self.stopwords = self._load_stopwords()
        
        # 初始化BERT模型(用于向量化)
        self.tokenizer = AutoTokenizer.from_pretrained('bert-base-chinese')
        self.model = AutoModel.from_pretrained('bert-base-chinese')
        self.model.eval()
    
    def extract_keywords_chinese(
        self, 
        text: str, 
        top_k: int = 20
    ) -> List[Dict]:
        """
        提取中文关键词
        
        Args:
            text: 文本内容
            top_k: 返回前K个关键词
            
        Returns:
            关键词列表,每个元素包含word和weight
        """
        # 方法1: TF-IDF
        tfidf_keywords = jieba.analyse.extract_tags(
            text,
            topK=top_k,
            withWeight=True
        )
        
        # 方法2: TextRank
        textrank_keywords = jieba.analyse.textrank(
            text,
            topK=top_k,
            withWeight=True
        )
        
        # 合并结果(取平均权重)
        keyword_dict = {}
        for word, weight in tfidf_keywords:
            keyword_dict[word] = keyword_dict.get(word, 0) + weight * 0.6
        
        for word, weight in textrank_keywords:
            keyword_dict[word] = keyword_dict.get(word, 0) + weight * 0.4
        
        # 排序并返回
        keywords = [
            {'word': word, 'weight': weight}
            for word, weight in sorted(
                keyword_dict.items(),
                key=lambda x: x[1],
                reverse=True
            )[:top_k]
        ]
        
        return keywords
    
    def extract_keywords_english(
        self,
        text: str,
        top_k: int = 20
    ) -> List[Dict]:
        """提取英文关键词"""
        from rake_nltk import Rake
        
        rake = Rake()
        rake.extract_keywords_from_text(text)
        keywords_with_scores = rake.get_ranked_phrases_with_scores()
        
        # 归一化分数
        max_score = max([score for score, _ in keywords_with_scores]) if keywords_with_scores else 1
        
        keywords = [
            {'word': phrase, 'weight': score / max_score}
            for score, phrase in keywords_with_scores[:top_k]
        ]
        
        return keywords
    
    def generate_embedding(self, text: str) -> np.ndarray:
        """
        生成文本向量
        
        Args:
            text: 输入文本
            
        Returns:
            768维向量
        """
        # 文本截断到512 tokens
        inputs = self.tokenizer(
            text,
            return_tensors='pt',
            max_length=512,
            truncation=True,
            padding=True
        )
        
        with torch.no_grad():
            outputs = self.model(**inputs)
            # 使用[CLS] token的输出作为句子向量
            embedding = outputs.last_hidden_state[:, 0, :].numpy()[0]
        
        # L2归一化
        embedding = embedding / np.linalg.norm(embedding)
        
        return embedding
    
    def calculate_similarity(
        self,
        text1: str,
        text2: str
    ) -> float:
        """计算两个文本的相似度"""
        emb1 = self.generate_embedding(text1)
        emb2 = self.generate_embedding(text2)
        
        # 余弦相似度
        similarity = np.dot(emb1, emb2)
        
        return float(similarity)
    
    def _load_stopwords(self) -> set:
        """加载停用词表"""
        # 简化版,实际应从文件加载
        stopwords = {
            '的', '了', '在', '是', '我', '有', '和', '就',
            '不', '人', '都', '一', '一个', '上', '也', '很',
            'the', 'a', 'an', 'and', 'or', 'but', 'in', 'on'
        }
        return stopwords
```

### 3.3 趋势分析模块

```python
# backend/services/trend_analyzer.py

from collections import Counter
from datetime import datetime, timedelta
from typing import List, Dict
import numpy as np
from scipy import stats

class TrendAnalyzer:
    """趋势分析器"""
    
    def analyze_keyword_trends(
        self,
        papers: List[Dict],
        time_window: str = 'year'
    ) -> Dict:
        """
        分析关键词趋势
        
        Args:
            papers: 文献列表
            time_window: 时间窗口(year, quarter, month)
            
        Returns:
            趋势分析结果
        """
        # 按时间分组统计关键词
        time_series = self._group_by_time(papers, time_window)
        
        # 计算每个关键词的趋势
        keyword_trends = {}
        all_keywords = set()
        
        for period, period_papers in time_series.items():
            keyword_counter = Counter()
            for paper in period_papers:
                for kw in paper.get('keywords', []):
                    keyword_counter[kw] += 1
                    all_keywords.add(kw)
            time_series[period] = dict(keyword_counter)
        
        # 为每个关键词生成时间序列
        for keyword in all_keywords:
            series = []
            periods = sorted(time_series.keys())
            
            for period in periods:
                count = time_series[period].get(keyword, 0)
                series.append(count)
            
            # 计算趋势指标
            trend_metrics = self._calculate_trend_metrics(series)
            
            keyword_trends[keyword] = {
                'series': series,
                'periods': [str(p) for p in periods],
                **trend_metrics
            }
        
        # 识别热点关键词(近期增长快)
        hot_keywords = self._identify_hot_keywords(keyword_trends)
        
        # 识别冷门关键词(近期下降快)
        cold_keywords = self._identify_cold_keywords(keyword_trends)
        
        return {
            'keyword_trends': keyword_trends,
            'hot_keywords': hot_keywords,
            'cold_keywords': cold_keywords,
            'time_window': time_window
        }
    
    def predict_future_trends(
        self,
        keyword_trends: Dict,
        forecast_periods: int = 4
    ) -> Dict:
        """
        预测未来趋势
        
        Args:
            keyword_trends: 历史趋势数据
            forecast_periods: 预测期数
            
        Returns:
            预测结果
        """
        predictions = {}
        
        for keyword, trend_data in keyword_trends.items():
            series = np.array(trend_data['series'])
            
            if len(series) < 3:
                continue  # 数据点太少,不预测
            
            # 简单线性回归预测
            x = np.arange(len(series))
            slope, intercept, r_value, _, _ = stats.linregress(x, series)
            
            # 预测未来值
            future_x = np.arange(len(series), len(series) + forecast_periods)
            predictions[keyword] = {
                'predicted_values': [
                    max(0, slope * x + intercept) for x in future_x
                ],
                'confidence': r_value ** 2,  # R²作为置信度
                'trend': 'up' if slope > 0 else 'down'
            }
        
        return predictions
    
    def _group_by_time(
        self,
        papers: List[Dict],
        window: str
    ) -> Dict:
        """按时间窗口分组"""
        grouped = {}
        
        for paper in papers:
            date = paper.get('publication_date') or paper.get('created_at')
            if not date:
                continue
            
            if isinstance(date, str):
                date = datetime.fromisoformat(date.replace('Z', '+00:00'))
            
            if window == 'year':
                period = date.year
            elif window == 'quarter':
                period = (date.year, (date.month - 1) // 3 + 1)
            elif window == 'month':
                period = (date.year, date.month)
            else:
                period = date.year
            
            if period not in grouped:
                grouped[period] = []
            grouped[period].append(paper)
        
        return grouped
    
    def _calculate_trend_metrics(self, series: List[int]) -> Dict:
        """计算趋势指标"""
        if len(series) < 2:
            return {
                'total_count': sum(series),
                'avg_count': np.mean(series),
                'growth_rate': 0,
                'volatility': 0
            }
        
        # 总出现次数
        total_count = sum(series)
        
        # 平均值
        avg_count = np.mean(series)
        
        # 增长率(最近值相对平均值)
        growth_rate = (series[-1] - avg_count) / (avg_count + 1) * 100
        
        # 波动性(标准差/平均值)
        volatility = np.std(series) / (avg_count + 1)
        
        return {
            'total_count': int(total_count),
            'avg_count': float(avg_count),
            'growth_rate': float(growth_rate),
            'volatility': float(volatility)
        }
    
    def _identify_hot_keywords(
        self,
        keyword_trends: Dict,
        top_n: int = 10
    ) -> List[Dict]:
        """识别热点关键词"""
        # 按增长率排序
        hot = sorted(
            keyword_trends.items(),
            key=lambda x: x[1]['growth_rate'],
            reverse=True
        )[:top_n]
        
        return [
            {
                'keyword': kw,
                'growth_rate': data['growth_rate'],
                'total_count': data['total_count']
            }
            for kw, data in hot
        ]
    
    def _identify_cold_keywords(
        self,
        keyword_trends: Dict,
        top_n: int = 10
    ) -> List[Dict]:
        """识别冷门关键词"""
        cold = sorted(
            keyword_trends.items(),
            key=lambda x: x[1]['growth_rate']
        )[:top_n]
        
        return [
            {
                'keyword': kw,
                'growth_rate': data['growth_rate'],
                'total_count': data['total_count']
            }
            for kw, data in cold
        ]
```
