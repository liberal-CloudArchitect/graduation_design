# 文献分析大数据平台 - 混合方案开发计划

## 🎯 方案概述

**选择方案**: 混合方案 (传统NLP + 大模型微调)

**核心理念**: 在关键功能上使用大模型微调展示技术深度,非核心功能使用传统方法控制成本

### 技术方案映射

| 功能模块 | 技术方案 | 理由 |
|---------|---------|------|
| **功能1: PDF解析** | ChatGLM3-6B 微调 ✅ | 核心功能,需要高准确率 |
| **功能2: 关键词提取** | BERT-base-chinese 微调 ✅ | 核心功能,展示NLP能力 |
| **功能3: 趋势分析** | ChatGLM3-6B 微调 ✅ | 核心功能,生成分析报告 |
| **功能4: 知识图谱** | 传统方法 + UIE ⭕ | 非核心,降低成本 |

### 关键指标

- **总投资**: 3万元 (相比完整方案节省50%)

### Week 13-14: NLP分析功能 - 传统方法

**关键词提取 - 传统方法**:
```python
# services/nlp_analyzer_traditional.py

import jieba
import jieba.analyse
from sklearn.feature_extraction.text import TfidfVectorizer

class TraditionalNLPAnalyzer:
    """传统NLP分析器"""
    
    def extract_keywords_tfidf(self, text: str, top_k: int = 20) -> list:
        """TF-IDF关键词提取"""
        keywords = jieba.analyse.extract_tags(
            text,
            topK=top_k,
            withWeight=True
        )
        
        return [
            {'word': word, 'weight': weight, 'method': 'tfidf'}
            for word, weight in keywords
        ]
    
    def extract_keywords_textrank(self, text: str, top_k: int = 20) -> list:
        """TextRank关键词提取"""
        keywords = jieba.analyse.textrank(
            text,
            topK=top_k,
            withWeight=True
        )
        
        return [
            {'word': word, 'weight': weight, 'method': 'textrank'}
            for word, weight in keywords
        ]
    
    def extract_keywords_hybrid(self, text: str, top_k: int = 20) -> list:
        """混合方法(TF-IDF + TextRank)"""
        tfidf_kws = self.extract_keywords_tfidf(text, top_k)
        textrank_kws = self.extract_keywords_textrank(text, top_k)
        
        # 合并并去重
        keyword_dict = {}
        for kw in tfidf_kws:
            keyword_dict[kw['word']] = keyword_dict.get(kw['word'], 0) + kw['weight'] * 0.6
        for kw in textrank_kws:
            keyword_dict[kw['word']] = keyword_dict.get(kw['word'], 0) + kw['weight'] * 0.4
        
        # 排序
        keywords = sorted(
            keyword_dict.items(),
            key=lambda x: x[1],
            reverse=True
        )[:top_k]
        
        return [
            {'word': word, 'weight': weight, 'method': 'hybrid'}
            for word, weight in keywords
        ]
```

```python
# scripts/download_arxiv.py
import arxiv
import os

def download_papers(category='cs.CL', max_results=5000):
    """从arXiv下载论文"""
    search = arxiv.Search(
        query=f'cat:{category}',
        max_results=max_results,
        sort_by=arxiv.SortCriterion.SubmittedDate
    )
    
    os.makedirs('data/raw_pdfs', exist_ok=True)
    
    for i, result in enumerate(search.results()):
        try:
            pdf_path = f'data/raw_pdfs/{result.get_short_id()}.pdf'
            result.download_pdf(filename=pdf_path)
            
            # 保存元数据
            metadata = {
                'title': result.title,
                'authors': [a.name for a in result.authors],
                'abstract': result.summary,
                'published': str(result.published),
                'categories': result.categories
            }
            
            print(f"Downloaded {i+1}/{max_results}: {result.title}")
            
        except Exception as e:
            print(f"Error downloading {result.get_short_id()}: {e}")

if __name__ == '__main__':
```

```python
# services/model_router.py

from enum import Enum
from typing import Dict, Any
import torch

class ModelType(Enum):
    PDF_PARSER_TRAD = "pdf_parser_traditional"  # 传统方法
    PDF_PARSER_LLM = "pdf_parser_llm"  # 大模型
    KEYWORD_TRAD = "keyword_traditional"
    KEYWORD_LLM = "keyword_llm"
    TREND_TRAD = "trend_traditional"
    TREND_LLM = "trend_llm"

class ModelRouter:
    """模型路由器 - 根据配置选择传统方法或大模型"""
    
    def __init__(self, use_llm: bool = True):
        self.use_llm = use_llm
        
        # 加载传统模型
        self.trad_pdf_parser = TraditionalPDFParser()
        self.trad_nlp_analyzer = TraditionalNLPAnalyzer()
        self.trad_trend_analyzer = TraditionalTrendAnalyzer()
        
        # 加载大模型 (如果启用)
        if use_llm:
            self._load_llm_models()
    
    def _load_llm_models(self):
        """加载大模型"""
        from transformers import AutoTokenizer, AutoModel
        
        print("加载大模型...")
        
        # 1. ChatGLM3 (PDF解析)
        self.chatglm_pdf_tokenizer = AutoTokenizer.from_pretrained(
            "models/chatglm3-pdf-parser-int8",
            trust_remote_code=True
        )
        self.chatglm_pdf_model = AutoModel.from_pretrained(
            "models/chatglm3-pdf-parser-int8",
            trust_remote_code=True,
            device_map="auto"
        ).eval()
        
        # 2. BERT (关键词)
        from transformers import BertForTokenClassification
        self.bert_keyword_tokenizer = AutoTokenizer.from_pretrained(
            "models/bert-keyword-extractor/final"
        )
        self.bert_keyword_model = BertForTokenClassification.from_pretrained(
            "models/bert-keyword-extractor/final"
        ).cuda().eval()
        
        # 3. ChatGLM3 (趋势分析)
        self.chatglm_trend_tokenizer = AutoTokenizer.from_pretrained(
            "models/chatglm3-trend-analyzer-int8",
            trust_remote_code=True
        )
        self.chatglm_trend_model = AutoModel.from_pretrained(
            "models/chatglm3-trend-analyzer-int8",
            trust_remote_code=True,
            device_map="auto"
        ).eval()
        
        print("✅ 大模型加载完成")
    
    async def parse_pdf(self, pdf_path: str) -> Dict:
        """PDF解析 - 路由到传统方法或大模型"""
        if self.use_llm:
            return await self._parse_pdf_llm(pdf_path)
        else:
            return self.trad_pdf_parser.parse_pdf(pdf_path)
    
    async def _parse_pdf_llm(self, pdf_path: str) -> Dict:
        """使用ChatGLM3解析PDF"""
        # 1. 先用传统方法提取文本
        text = self.trad_pdf_parser._parse_text_pdf(pdf_path)['full_text']
        
        # 2. 使用ChatGLM3提取元数据
        prompt = f"""请从以下学术论文文本中提取元数据,以JSON格式返回。

论文文本:
{text[:2000]}

请提取:
1. title: 论文标题
2. authors: 作者列表
3. abstract: 摘要
4. keywords: 关键词列表

返回JSON:"""
        
        response, _ = self.chatglm_pdf_model.chat(
            self.chatglm_pdf_tokenizer,
            prompt,
            history=[]
        )
        
        # 3. 解析JSON
        import json
        try:
            metadata = json.loads(response)
            metadata['method'] = 'chatglm3'
            return metadata
        except:
            # 如果解析失败,降级到传统方法
            return self.trad_pdf_parser.parse_pdf(pdf_path)
    
    async def extract_keywords(self, text: str, top_k: int = 20) -> list:
        """关键词提取 - 路由"""
        if self.use_llm:
            return await self._extract_keywords_llm(text, top_k)
        else:
            return self.trad_nlp_analyzer.extract_keywords_hybrid(text, top_k)
    
    async def _extract_keywords_llm(self, text: str, top_k: int = 20) -> list:
        """使用BERT提取关键词"""
        # Tokenize
        inputs = self.bert_keyword_tokenizer(
            text,
            return_tensors='pt',
            max_length=512,
            truncation=True,
            padding=True
        ).to('cuda')
        
        # 预测
        with torch.no_grad():
            outputs = self.bert_keyword_model(**inputs)
            predictions = torch.argmax(outputs.logits, dim=2)
        
        # 解码关键词
        keywords = self._decode_keywords(
            inputs['input_ids'][0],
            predictions[0]
        )
        
        return keywords[:top_k]
    
    def _decode_keywords(self, input_ids, predictions):
        """从BIO标注解码关键词"""
        tokens = self.bert_keyword_tokenizer.convert_ids_to_tokens(input_ids)
        label_list = ["O", "B-KW", "I-KW"]
        
        keywords = []
        current_keyword = []
        
        for token, pred in zip(tokens, predictions):
            label = label_list[pred.item()]
            
            if label == 'B-KW':  # 关键词开始
                if current_keyword:
                    keywords.append({
                        'word': ''.join(current_keyword),
                        'weight': 1.0,
                        'method': 'bert'
                    })
                current_keyword = [token.replace('##', '')]
            elif label == 'I-KW':  # 关键词内部
                current_keyword.append(token.replace('##', ''))
            else:  # O
                if current_keyword:
                    keywords.append({
                        'word': ''.join(current_keyword),
                        'weight': 1.0,
                        'method': 'bert'
                    })
                    current_keyword = []
        
        return keywords

# 全局模型路由器
model_router = ModelRouter(use_llm=True)
```

```python
    """计算准确率"""
    correct = 0
    total = 0
    
    for result, truth in zip(results, ground_truth):
        # 标题匹配
        if result.get('title') == truth['ground_truth']['title']:
            correct += 1
        total += 1
    
    return correct / total
```

### 可能的提问

**Q1: 为什么不全部使用大模型?**
A: 成本考虑。混合方案在保证核心功能准确率的同时,控制成本。知识图谱功能使用传统方法+UIE已经足够,节省训练时间和GPU资源。

**Q2: 模型训练数据从哪里来?**
A: 1) arXiv公开论文5000篇 2) 众包标注平台标注元数据 3) 半自动化标注(GPT-4生成+人工审核)

**Q3: 如何保证模型效果?**
A: 1) 高质量训练数据 2) 合适的LoRA参数 3) 充分的训练epoch 4) 验证集监控 5) A/B对比测试

**Q4: 推理速度慢怎么办?**
A: 1) 模型量化(INT8) 2) 缓存常见查询 3) 异步处理 4) GPU加速 5) 批处理优化

**Q5: 大模型部署成本高吗?**
A: 使用量化模型后,单卡RTX 4090(24GB)即可部署,月成本¥8000左右。或使用CPU推理,速度慢但成本更低。
