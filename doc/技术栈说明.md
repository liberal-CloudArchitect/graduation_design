```typescript
import axios from 'axios';

const api = axios.create({
  baseURL: import.meta.env.VITE_API_URL,

});

// 请求拦截器
api.interceptors.request.use(
  (config) => {
    const token = localStorage.getItem('token');
    if (token) {
      config.headers.Authorization = `Bearer ${token}`;
    }
    return config;
  },
  (error) => Promise.reject(error)
);

// 响应拦截器
api.interceptors.response.use(
  (response) => response.data,
  (error) => {

      window.location.href = '/auth/login';
    }
    return Promise.reject(error);
  }
);
```

```python
class PaperCreate(BaseModel):
```

```python
from celery import Celery
from app.core.config import settings

celery_app = Celery(
    "paper_bigdata",
    broker=settings.CELERY_BROKER_URL,
    backend=settings.CELERY_RESULT_BACKEND
)

@celery_app.task(bind=True, max_retries=3)
def parse_pdf_task(self, paper_id: int):
    """异步解析PDF"""
    try:
        # 1. 下载PDF文件
        pdf_path = download_pdf(paper_id)
        
        # 2. 解析PDF
        parser = PDFParser()
        result = parser.parse(pdf_path)
        
        # 3. 存储结果
        save_parse_result(paper_id, result)
        
        return {"status": "completed", "paper_id": paper_id}
        
    except Exception as exc:
        # 重试
        raise self.retry(exc=exc, countdown=60)
```

## 技术选型总结

### 技术栈优势

1. **前后端分离**: React + FastAPI,开发效率高
2. **类型安全**: TypeScript + Pydantic,减少bug
3. **高性能**: 异步处理 + 缓存策略
4. **可扩展**: 微服务架构 + 容器化部署
5. **AI能力**: BERT + NLP算法
6. **可视化**: ECharts + G6,展示效果好
7. **开发体验**: 热更新 + 自动文档生成

### 技术风险与应对

| 风险 | 应对措施 |
|------|---------|
| BERT模型计算资源需求大 | 使用轻量级模型,GPU加速 |
| 大文件上传处理 | 分片上传,异步处理 |
| 数据库性能瓶颈 | 索引优化,读写分离,缓存 |
| 前端打包体积大 | 代码分割,懒加载,CDN |
