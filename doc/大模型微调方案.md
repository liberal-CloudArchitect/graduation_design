# 开源大模型微调方案 - 四模型专用化架构

## 📋 方案概述

### 核心思想
将系统的四个主要功能分别使用**不同的开源大模型进行微调**，实现专用化优化，提升各功能模块的性能和准确度。

### 创新点
- ✨ **多模型协同**: 4个专用模型并行工作
- ✨ **任务特化**: 每个模型针对特定任务优化
- ✨ **性能优化**: 选择最适合的模型架构
- ✨ **技术前沿**: 展示大模型微调能力

# services/model_router.py
from enum import Enum
from typing import Dict, Any

class ModelType(Enum):
    PDF_PARSER = "pdf_parser"
    KEYWORD_EXTRACTOR = "keyword_extractor"
    TREND_ANALYZER = "trend_analyzer"
    KG_BUILDER = "kg_builder"

class ModelRouter:
    """模型路由器"""
    
    def __init__(self):
        # 初始化各个模型
        self.models = {
            ModelType.PDF_PARSER: ChatGLMPDFParser(),
            ModelType.KEYWORD_EXTRACTOR: BERTKeywordExtractor(),
            ModelType.TREND_ANALYZER: ChatGLMTrendAnalyzer(),
            ModelType.KG_BUILDER: UIEKGBuilder()
        }
    
    async def route(
        self,
        task_type: ModelType,
        input_data: Dict[str, Any]
    ) -> Dict[str, Any]:
        """路由到对应模型"""
        model = self.models.get(task_type)
        if not model:
            raise ValueError(f"Unknown model type: {task_type}")
        
        result = await model.predict(input_data)
        return result

# services/bert_keyword_extractor.py
class BERTKeywordExtractor:
    """BERT关键词提取模型"""
    
    def __init__(self):
        from transformers import BertForTokenClassification, BertTokenizer
        model_path = "models/bert-keyword-extractor"  # 微调后的模型
        self.tokenizer = BertTokenizer.from_pretrained(model_path)
        self.model = BertForTokenClassification.from_pretrained(model_path)
        self.model.cuda()
        self.model.eval()
    
    async def predict(self, input_data: Dict) -> Dict:
        """预测关键词"""
        text = input_data['text']
        
        # Tokenize
        inputs = self.tokenizer(
            text,
            return_tensors='pt',
            max_length=512,
            truncation=True,
            padding=True
        ).to('cuda')
        
        # 预测
        with torch.no_grad():
            outputs = self.model(**inputs)
            predictions = torch.argmax(outputs.logits, dim=2)
        
        # 解码关键词
        keywords = self._decode_keywords(
            inputs['input_ids'][0],
            predictions[0]
        )
        
        return {'keywords': keywords}
    
    def _decode_keywords(self, input_ids, predictions):
        """从BIO标注中解码关键词"""
        tokens = self.tokenizer.convert_ids_to_tokens(input_ids)
        keywords = []
        current_keyword = []
        
        for token, pred in zip(tokens, predictions):
            label = self.id2label[pred.item()]
            
            if label == 'B':  # Begin
                if current_keyword:
                    keywords.append(''.join(current_keyword))
                current_keyword = [token]
            elif label == 'I':  # Inside
                current_keyword.append(token)
            else:  # O (Outside)
                if current_keyword:
                    keywords.append(''.join(current_keyword))
                    current_keyword = []
        
        return keywords

        {'text': metadata['abstract']}
    )
    
    # 4. 合并结果
    metadata['keywords'] = keywords['keywords']
    
    return metadata
