### Week 7-8: RAG引擎骨架

```python
# app/rag/engine.py
from FlagEmbedding import BGEM3FlagModel
from pymilvus import MilvusClient

class RAGEngine:
    """RAG引擎核心"""
    
    def __init__(self):
        self.embedder = BGEM3FlagModel('BAAI/bge-m3')
        self.milvus = MilvusClient(uri="http://localhost:19530")
        self.collection_name = "papers"
    
    async def index_paper(self, paper_id: int, chunks: List[str]):
        """索引文献到向量库"""
        embeddings = self.embedder.encode(chunks)['dense_vecs']
        
        entities = [
            {"id": f"{paper_id}_{i}", "vector": emb, "text": chunk, "paper_id": paper_id}
            for i, (emb, chunk) in enumerate(zip(embeddings, chunks))
        ]
        
        self.milvus.insert(self.collection_name, entities)
    
    async def search(self, query: str, top_k: int = 5) -> List[dict]:
        """向量检索"""
        query_embedding = self.embedder.encode(query)['dense_vecs'][0]
        
        results = self.milvus.search(
            self.collection_name,
            data=[query_embedding],
            limit=top_k,
            output_fields=["text", "paper_id"]
        )
        
        return results[0]
```

```python
# app/api/v1/papers.py
from fastapi import APIRouter, UploadFile, Depends
from app.services.pdf_parser import PDFParser
from app.rag.engine import RAGEngine

router = APIRouter()
pdf_parser = PDFParser()
rag_engine = RAGEngine()

@router.post("/papers")
async def upload_paper(
    file: UploadFile,
    current_user: User = Depends(get_current_user)
):
    # 1. 保存文件
    file_path = await save_file(file)
    
    # 2. 解析PDF
    parsed = await pdf_parser.parse(file_path)
    
    # 3. 存储到数据库
    paper = Paper(
        user_id=current_user.id,
        title=parsed['title'],
        authors=parsed['authors'],
        abstract=parsed['abstract']
    )
    db.add(paper)
    db.commit()
    
    # 4. 索引到向量库 (异步任务)
    index_paper_task.delay(paper.id, parsed['chunks'])
    
    return {"id": paper.id, "status": "processing"}
```

### Week 13-16: RAG问答API

```python
# app/api/v1/rag.py
from fastapi import APIRouter
from fastapi.responses import StreamingResponse
from app.rag.engine import RAGEngine
from app.rag.generator import RAGGenerator

router = APIRouter()
rag_engine = RAGEngine()
generator = RAGGenerator()

@router.post("/rag/ask")
async def ask_question(request: QuestionRequest):
    """同步问答"""
    # 1. 检索
    docs = await rag_engine.search(request.question, top_k=5)
    
    # 2. 生成
    answer = await generator.generate(request.question, docs)
    
    return {
        "answer": answer,
        "references": [{"text": d["text"], "paper_id": d["paper_id"]} for d in docs]
    }

@router.get("/rag/stream")
async def stream_answer(question: str):
    """流式问答 (SSE)"""
    async def generate():
        docs = await rag_engine.search(question)
        async for chunk in generator.stream_generate(question, docs):
            yield f"data: {chunk}\n\n"
    
    return StreamingResponse(generate(), media_type="text/event-stream")
```

```python
# app/external/semantic_scholar.py
import httpx

class SemanticScholarClient:
    BASE_URL = "https://api.semanticscholar.org/graph/v1"
    
    async def search(self, query: str, limit: int = 10):
        async with httpx.AsyncClient() as client:
            response = await client.get(
                f"{self.BASE_URL}/paper/search",
                params={"query": query, "limit": limit,
                        "fields": "title,abstract,authors,year,citationCount"}
            )
            return response.json()['data']

# app/external/openalex.py
class OpenAlexClient:
    BASE_URL = "https://api.openalex.org"
    
    async def search_works(self, query: str):
        async with httpx.AsyncClient() as client:
            response = await client.get(
                f"{self.BASE_URL}/works",
                params={"search": query, "per_page": 20}
            )
            return response.json()['results']
```
