### Week 7-8: 基础组件库

```typescript
// components/Chat/ChatWindow.tsx
import { useState } from 'react';
import { Input, Button, List, Avatar, Spin } from 'antd';

interface Message {
  role: 'user' | 'assistant';
  content: string;
  references?: Reference[];
}

export const ChatWindow: React.FC = () => {
  const [messages, setMessages] = useState<Message[]>([]);
  const [loading, setLoading] = useState(false);
  
  const handleSend = async (question: string) => {
    setLoading(true);
    // 调用RAG API
    const response = await ragService.ask(question);
    setMessages(prev => [...prev, 
      { role: 'user', content: question },
      { role: 'assistant', content: response.answer, references: response.refs }
    ]);
    setLoading(false);
  };
  
  return (
    <div className="chat-window">
      <List dataSource={messages} renderItem={...} />
      <Input.Search onSearch={handleSend} loading={loading} />
    </div>
  );
};
```

  // 支持流式输出
  const [text, setText] = useState('');
  
  useEffect(() => {
    const eventSource = new EventSource('/api/rag/stream');
    eventSource.onmessage = (e) => setText(prev => prev + e.data);
    return () => eventSource.close();
  }, []);
  
  return <Markdown>{text}</Markdown>;

### Week 25-32: Agent界面

```typescript
// pages/Agent/LiteratureReview.tsx
export const LiteratureReviewPage: React.FC = () => {
  const [topic, setTopic] = useState('');
  const [progress, setProgress] = useState<AgentProgress>({ step: 0 });
  
  const handleGenerate = async () => {
    // 展示Agent执行进度
    const stream = agentService.generateReview(topic);
    for await (const update of stream) {
      setProgress(update);
    }
  };
  
  return (
    <div>
      <Input value={topic} onChange={e => setTopic(e.target.value)} />
      <AgentProgressView progress={progress} />
      <Button onClick={handleGenerate}>生成综述</Button>
    </div>
  );
};
```
